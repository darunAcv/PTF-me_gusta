const like=document.getElementsByTagName('button');
for (let i= 0; i<like.length; i++){
    like[i].addEventListener('click', function(){
        this.previousElementSibling.getElementsByClassName('counter').innerText= parseInt(this.previousElementSibling.getElementsByClassName('counter').innerText) +1;
    });
}
